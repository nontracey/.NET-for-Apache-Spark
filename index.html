using Microsoft.Spark.Sql;

namespace MySparkApp
{
    /// <summary>
    /// spark-submit --class org.apache.spark.deploy.dotnet.DotnetRunner --master local bin\Debug\net6.0\microsoft-spark-3-2_2.12-2.1.1.jar dotnet bin\Debug\net6.0\mySparkApp.dll
    /// </summary>
    class Program
    {
        static void Main(string[] args)
        {
            // Create a Spark session/Job
            SparkSession spark = SparkSession
                .Builder()
                .AppName("word_count_sample")
                .GetOrCreate();

            // Create initial DataFrame
            DataFrame dataFrame = spark.Read().Text("input.txt");

            // Count words
            DataFrame words = dataFrame
                .Select(Functions.Split(Functions.Col("value"), " ").Alias("words"))
                .Select(Functions.Explode(Functions.Col("words"))
                .Alias("word"))
                .GroupBy("word")
                .Count()
                .OrderBy(Functions.Col("count").Desc());

            // Show results
            words.Show();

            // Stop Spark session
            spark.Stop();
        }
    }
}

using System;
using Microsoft.Data.Analysis;
using Microsoft.Spark.Sql;
using Microsoft.Spark.Sql.Types;
using DataFrame = Microsoft.Spark.Sql.DataFrame;
using FxDataFrame = Microsoft.Data.Analysis.DataFrame;
using StructType = Microsoft.Spark.Sql.Types.StructType;

namespace VectorUdfSample.Test
{
    /// <summary>
    /// Demo1
    /// spark-submit --class org.apache.spark.deploy.dotnet.DotnetRunner --master local D:\Project\SparkProject\NET.ApacheSpark.Lesson\VectorUdfSamples\bin\Debug\net6.0\microsoft-spark-3-2_2.12-2.1.1.jar dotnet D:\Project\SparkProject\NET.ApacheSpark.Lesson\VectorUdfSamples\bin\Debug\net6.0\VectorUdfSamples.dll
    /// </summary>
    class Program
    {
        public static void Main(string[] args)
        {

            SparkSession spark = SparkSession
                .Builder()
                // shuffle
                .Config("spark.sql.shuffle.partitions", "3")
                .AppName("SQL VectorUdfs example using .NET for Apache Spark")
                .GetOrCreate();

            DataFrame df = spark.Read().Schema("age INT, name STRING").Json("people.json");

            StructType schema = df.Schema();
            Console.WriteLine(schema.SimpleString);

            df.Show();

            df.PrintSchema();

            // 
            df.GroupBy("age")
                .Apply(
                    new StructType(new[]
                    {
                        new StructField("age", new IntegerType()),
                        new StructField("nameCharCount", new IntegerType())
                    }),
                    r => CountCharacters(r))
                .Show();
            spark.Stop();
            Console.ReadLine();
        }

        private static FxDataFrame CountCharacters(FxDataFrame dataFrame)
        {
            int characterCount = 0;

            var characterCountColumn = new Int32DataFrameColumn("nameCharCount");
            var ageColumn = new Int32DataFrameColumn("age");
            ArrowStringDataFrameColumn nameColumn = dataFrame.Columns.GetArrowStringColumn("name");
            for (long i = 0; i < dataFrame.Rows.Count; ++i)
            {
                characterCount += nameColumn[i].Length;
            }

            if (dataFrame.Rows.Count > 0)
            {
                characterCountColumn.Append(characterCount);
                ageColumn.Append(dataFrame.Columns.GetInt32Column("age")[0]);
            }

            return new FxDataFrame(ageColumn, characterCountColumn);
        }
    }
}

private static void JaccardTest()
{
    // SparkSession
    SparkSession session = SparkSession.Builder()
        .AppName("ItemCFApp")
        .GetOrCreate();

    //jdbcDf.Show();
    // 
    DataFrame projectsDf = session
        .Read()
        .Schema("id INT, user_id INT, item_id INT")
        .Csv("user_item.csv");
    projectsDf.Show();

    // 
    var itemCount = projectsDf.GroupBy("item_id").Count();
    itemCount.Show();
    // 
    var item2ItemCount = projectsDf.As("a")
        .Join(projectsDf.As("b"), Col("a.user_id") == Col("b.user_id"))
        .Where(Col("a.item_id").NotEqual(Col("b.item_id")))
        .Select(Col("a.item_id").As("a_item_id"), Col("b.item_id").As("b_item_id"))
        .GroupBy("a_item_id", "b_item_id").Count();

    item2ItemCount.Show();
    // 
    var result = item2ItemCount.As("i2i")
        .Join(itemCount.As("ic1"), Col("i2i.a_item_id") == Col("ic1.item_id"))
        .Join(itemCount.As("ic2"), Col("i2i.b_item_id") == Col("ic2.item_id"))
        .SelectExpr("i2i.a_item_id", "i2i.b_item_id", "i2i.count/(ic1.count + ic2.count - i2i.count) as similarity");
    result.Show();

    session.Stop();
}
